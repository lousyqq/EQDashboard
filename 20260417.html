function renderSidebarMenus() { if (!currentUser) return; const currentFabObj = getFabs().find(f => f.fabName === currentFab); if (!currentFabObj) return; let menus = JSON.parse(JSON.stringify(getCustomMenus())); let pSets = currentLayoutMode === 'personal' ? getPersonalSettings(currentUser.id) : {}; const fabRoleIds = currentFabObj.assignedRoles || []; const userRoleIds = currentUser.assignedRoles || []; const activeRoleIds = (currentUser.roleLevel === 'admin') ? fabRoleIds : fabRoleIds.filter(id => userRoleIds.includes(id)); const roles = getRoles(); let initialMenuIds = []; activeRoleIds.forEach(roleId => { const role = roles.find(r => r.id === roleId); if (role && role.allowedMenuIds) { role.allowedMenuIds.forEach(mId => { if (!initialMenuIds.includes(mId)) initialMenuIds.push(mId); }); } }); let allowedIds = getAllowedIdsWithHierarchy(menus, initialMenuIds); if (currentLayoutMode === 'personal') { menus.forEach(m => { if(pSets[m.id]) { if(pSets[m.id].hidden !== undefined) m.enabled = !pSets[m.id].hidden; if(pSets[m.id].target !== undefined) m.target = pSets[m.id].target; if(pSets[m.id].order !== undefined) m.order = pSets[m.id].order; } }); } menus = menus.filter(m => m.enabled !== false && allowedIds.has(m.id)); menus.sort((a, b) => { if (currentLayoutMode === 'system') { if ((!a.parentId && (!a.parentIds || a.parentIds.length === 0)) && (!b.parentId && (!b.parentIds || b.parentIds.length === 0))) { let idxA = initialMenuIds.indexOf(a.id); let idxB = initialMenuIds.indexOf(b.id); return (idxA === -1 ? 9999 : idxA) - (idxB === -1 ? 9999 : idxB); } return (a.order || 0) - (b.order || 0); } return (a.order || 0) - (b.order || 0); }); const rootMenus = menus.filter(m => m.isPoolItem === false && !m.parentId && (!m.parentIds || m.parentIds.length === 0)); // 防呆：如果沒有目前選中的主選單，預設選取第一個 if ((!currentActiveTopMenuId || !rootMenus.find(m => m.id === currentActiveTopMenuId)) && rootMenus.length > 0) { currentActiveTopMenuId = rootMenus[0].id; } // ================= 1. 渲染上方主選單 ================= let topLinksHtml = ''; rootMenus.forEach(root => { let dName = root.displayName; if (typeof i18n !== 'undefined' && i18n[currentLang] && i18n[currentLang]['dyn_' + root.id] && !root.isEdited) { dName = i18n[currentLang]['dyn_' + root.id]; } const isActive = (root.id === currentActiveTopMenuId) ? 'active' : ''; const actionAttr = `onclick="selectTopMenu('${root.id}')"`; topLinksHtml += `${dName}`; }); const topMenusContainer = document.getElementById('top-dynamic-menus'); if (topMenusContainer) topMenusContainer.innerHTML = topLinksHtml; // ================= 2. 渲染左側系統看板 (工程師 UI 版) ================= let html = ''; const activeRoot = rootMenus.find(m => m.id === currentActiveTopMenuId); if (activeRoot) { let dName = activeRoot.displayName; if (typeof i18n !== 'undefined' && i18n[currentLang] && i18n[currentLang]['dyn_' + activeRoot.id] && !activeRoot.isEdited) { dName = i18n[currentLang]['dyn_' + activeRoot.id]; } // 產生深色的頂部標題 (Header) const iconHtml = generateIconHtml(activeRoot.icon, '', '', activeRoot.menuMode === 'folder'); html += `
${iconHtml} ${dName}
`; // 產生其底下展開的子選單 const children = menus.filter(m => m.parentId === activeRoot.id || (m.parentIds && m.parentIds.includes(activeRoot.id))); children.sort((a,b) => (a.parentOrders?.[activeRoot.id] ?? a.order ?? 0) - (b.parentOrders?.[activeRoot.id] ?? b.order ?? 0)); children.forEach(child => { html += generateSidebarMenuItem(child, menus, 1, false); }); } document.getElementById('dynamic-sidebar-menus').innerHTML = html; }
