function generateSidebarMenuItem(menu, allMenus, level, forceExpand = false) { const children = allMenus.filter(m => m.parentId === menu.id || (m.parentIds && m.parentIds.includes(menu.id))); children.sort((a,b) => (a.parentOrders?.[menu.id] ?? a.order ?? 0) - (b.parentOrders?.[menu.id] ?? b.order ?? 0)); const hasChildren = children.length > 0; let iconHtml = generateIconHtml(menu.icon, '', '', menu.menuMode === 'folder'); let actionAttr = ''; if (hasChildren) actionAttr = `data-bs-toggle="collapse" data-bs-target="#collapse-${menu.id.replace(/\./g, '_')}"`; else if (menu.menuMode === 'app_grid') actionAttr = `onclick="openAppGridPage('${menu.id}', '${menu.displayName}', this)"`; else if (menu.url) { if (menu.target === 'blank') actionAttr = `onclick="window.open('${menu.url}', '_blank')"` else if (menu.target === 'fullscreen') actionAttr = `onclick="openDynamicIframe('${menu.url}', '${menu.displayName}', this, true)"`; else actionAttr = `onclick="openDynamicIframe('${menu.url}', '${menu.displayName}', this, false)"`; } else if (menu.targetPage) actionAttr = `onclick="navTo('${menu.targetPage}', this, '${menu.displayName}')"`; let dName = menu.displayName; if (typeof i18n !== 'undefined' && i18n[currentLang] && i18n[currentLang]['dyn_' + menu.id] && !menu.isEdited) { dName = i18n[currentLang]['dyn_' + menu.id]; } // 判斷層級來套用不同的工程師介面樣式 if (level === 1) { if (hasChildren) { const expandClass = forceExpand ? 'show' : ''; const ariaAttr = forceExpand ? 'true' : 'false'; let html = `
${iconHtml} ${dName}
`; children.forEach(child => html += generateSidebarMenuItem(child, allMenus, level + 1, false)); html += `
`; return html; } else { return `
${iconHtml}${dName}
`; } } else if (level >= 2) { if (hasChildren) { const expandClass = forceExpand ? 'show' : ''; const ariaAttr = forceExpand ? 'true' : 'false'; let html = `
${iconHtml} ${dName}
`; children.forEach(child => html += generateSidebarMenuItem(child, allMenus, level + 1, false)); html += `
`; return html; } else { return `
${iconHtml}${dName}
`; } } return ''; }
